CREATE TABLE `users` (
	`userId` INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
	`firstname` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_unicode_ci',
	`lastname` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_unicode_ci',
	`username` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_unicode_ci',
	`password` VARCHAR(50) NOT NULL COLLATE 'utf8mb3_unicode_ci',
	`imgPath` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb3_unicode_ci',
	`updatedDate` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	`createdDate` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	PRIMARY KEY (`userId`) USING BTREE
)
COLLATE='utf8mb3_unicode_ci'
ENGINE=InnoDB
AUTO_INCREMENT=4
;
